Exemplos de Código

Uso Básico
// Importar o sistema import { LoadingUtils } from './unifiedLoadingManager.js'; // Mostrar loading const loadingId = LoadingUtils.show('CHAT_LOADING', { message: 'Carregando chat...' }); // Esconder loading LoadingUtils.hide(loadingId);
Loading com Progresso
const loadingId = LoadingUtils.show('FILE_PROCESSING', { message: 'Processando arquivo...', showProgress: true }); // Atualizar progresso LoadingUtils.progress(loadingId, 50, 'Processando... 50%');
Loading com Etapas
const loadingId = LoadingUtils.show('APP_INITIALIZATION', { steps: ['Etapa 1', 'Etapa 2', 'Etapa 3'] }); // Marcar etapas como concluídas LoadingUtils.step(loadingId, 'Etapa 1', 'completed'); LoadingUtils.step(loadingId, 'Etapa 2', 'loading');
